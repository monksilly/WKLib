56 private static WKLibAPI Create_Internal(string displayName, string guid, string defaultConfigFileName = "DefaultConfig")
57 {
58 foreach(WKLibAPI API in internalAPIs)
59 {
60 if (string.Equals(guid, API.GUID))
61 throw new Exception($"{displayName} collides with {API.DisplayName}, they both have the same guid, {guid}");
62 }
63
64 WKLibAPI newAPI = new WKLibAPI(displayName, guid, defaultConfigFileName);
65 internalAPIs.Add(newAPI);
