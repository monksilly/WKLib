202 public ConfigFolder LoadFiles()
203 {
204 // Load existing JSON files
205 string[] jsonFiles = Directory.GetFiles(BasePath, "*.json", SearchOption.TopDirectoryOnly);
206 foreach (var jsonFile in jsonFiles)
207 {
208 if (string.IsNullOrWhiteSpace(jsonFile))
209 continue;
210
211 RegisterConfigFile(Path.GetFileName(jsonFile));
212 }
213
214 string[] folders = Directory.GetDirectories(BasePath, "*", SearchOption.TopDirectoryOnly);
215 foreach (var folder in folders)
216 {
217 if (string.IsNullOrWhiteSpace(folder))
218 continue;
219
220 RegisterConfigFolder(folder);
221 }
222
223 return this;
