49 public async Task<AssetBundle> LoadBundleRelativeAsync(
50 string relativePath,
51 IProgress<float> progress = null)
52 {
53 progress?.Report(0f);
54 if (string.IsNullOrEmpty(relativePath))
55 {
56 WKLog.Error("[AssetService] Path is null or empty.");
57 progress?.Report(1f);
58 return null;
59 }
60
61 var cacheKey = $"{_wkLibAPI.DisplayName}:{Path.GetFileNameWithoutExtension(relativePath)}";
62 if (_bundleCache.TryGetValue(cacheKey, out var cached) && cached is not null)
63 {
64 WKLog.Debug($"[AssetService] Returning cached bundle: {cacheKey}");
65 progress?.Report(1f);
66 return cached;
67 }
68
69 var fullPath = Path.Combine(_assemblyFolder, relativePath);
70 if (!File.Exists(fullPath))
71 {
72 WKLog.Error($"[AssetService] Bundle not found at {fullPath}");
73 progress?.Report(1f);
74 return null;
75 }
76
77 var request = AssetBundle.LoadFromFileAsync(fullPath);
78 if (request is null)
79 {
80 WKLog.Error($"[AssetService] Failed to load bundle: {fullPath} (request)");
81 progress?.Report(1f);
82 return null;
83 }
84
85 try
86 {
87 while (!request.isDone)
88 {
89 _sQuitCts.Token.ThrowIfCancellationRequested();
90
91 progress?.Report(request.progress * 1.1f);
92 await Task.Yield();
93 }
94 }
95 catch (OperationCanceledException)
96 {
97 WKLog.Debug($"[AssetService] Asset Bundle Load cancelled");
98 return null;
99 }
100
101 var bundle = request.assetBundle;
102 if (bundle is null)
103 {
104 WKLog.Error($"[AssetService] Failed to load bundle at {fullPath} (bundle)");
105 progress?.Report(1f);
106 return null;
107 }
108
109 _bundleCache[cacheKey] = bundle;
110 WKLog.Debug($"[AssetService] Loaded & cached bundle: {cacheKey}");
111 progress?.Report(1f);
112 return bundle;
113 }

177 public async Task<Dictionary<string, M_Level>> LoadAllLevelsFromBundle(
178 AssetBundle bundle,
179 IProgress<float> progress = null)
180 {
181 var foundLevels = new Dictionary<string, M_Level>();
182
183 if (bundle is null)
184 {
185 WKLog.Error("[AssetService] LoadAllLevelsFromBundle called with null bundle.");
186 progress?.Report(1f);
187 return foundLevels;
188 }
189
190 var cacheKey = $"{_wkLibAPI.DisplayName}:{bundle.name}";
191
192 if (_loadedLevelsCache.ContainsKey(cacheKey))
193 {
194 WKLog.Debug($"[AssetService] Returning Cached loaded levels for bundle: {cacheKey}");
195 progress?.Report(1f);
196 return _loadedLevelsCache.GetValueOrDefault(cacheKey);
197 }
198
199 var request = bundle.LoadAllAssetsAsync<GameObject>();
200 try
201 {
202 while (!request.isDone)
203 {
204 _sQuitCts.Token.ThrowIfCancellationRequested();
205
206 progress?.Report(request.progress * 0.5f);
207 await Task.Yield();
208 }
209 }
210 catch (OperationCanceledException)
211 {
212 WKLog.Debug($"[AssetService] LoadAllLevelsFromBundle canceled.");
213 return null;
214 }
215
216
217 var currentDatabase = CL_AssetManager.GetDatabase(_wkLibAPI.DisplayName);
218 if (currentDatabase == null)
219 {
220 WKLog.Info($"[AssetService] Creating new asset database for {_wkLibAPI.DisplayName}");
221 currentDatabase = new WKAssetDatabase();
222
223 CL_AssetManager.AddNewDatabase(_wkLibAPI.DisplayName, currentDatabase);
224 }
225
226 var allGOs = request.allAssets.Cast<GameObject>().ToList();
227 var total = allGOs.Count;
228 const int batchSize = 20;
229
230 for (var i = 0; i < total; i += batchSize)
231 {
232 for (var j = i; j < i + batchSize && j < total; j++)
233 {
234 var go = allGOs[j];
235 if (go.TryGetComponent<M_Level>(out var level))
236 {
237 bool alreadyExists = false;
238 foreach (var levelAsset in currentDatabase.levelAssets)
239 {
240 if (levelAsset == null)
241 continue;
242
243 if (levelAsset.id == level.levelName)
244 {
245 alreadyExists = true;
246 break;
247 }
248 }
249
250 if (!alreadyExists)
251 currentDatabase.levelAssets.Add(M_Level.LevelAssetHolder.GetNewHolderFromLevel(level));
252
253 // Log its name for debugging
254 WKLog.Debug($"[AssetService] Found level: {level.name}");
255
256 if (!foundLevels.TryAdd(level.name, level))
257 WKLog.Debug($"[AssetService] Duplicate prefab.name '{level.name}' found; skipping the duplicate.");
258 }
259 }
260 progress?.Report(0.5f + (i / (float)total) * 0.5f);
261 await Task.Yield();
262 }
263
264 _loadedLevelsCache[cacheKey] = foundLevels;
265 progress?.Report(1f);
266 return foundLevels;
267 }

312 while (!request.isDone)
313 {
314 progress?.Report(request.progress);
315 await Task.Yield();
316 }
317
318 var gamemode = request.asset as M_Gamemode;
319
320 if (gamemode is null)
321 {
322 WKLog.Error($"[AssetService] Gamemode '{assetName}' not found in bundle {bundle.name}");
323 progress?.Report(1f);
324 return null;
325 }
326
327 if (string.IsNullOrEmpty(gamemode.unlockAchievement))
328 gamemode.unlockAchievement = "ACH_TUTORIAL";
329
330 gamemode.gamemodePanel = Resources.FindObjectsOfTypeAll<UI_GamemodeScreen_Panel>()
331 .FirstOrDefault(x => x.name == "Gamemode_Panel_Base");
332 gamemode.loseScreen = Resources.FindObjectsOfTypeAll<UI_ScoreScreen>()
333 .FirstOrDefault(x => x.name == "ScorePanel_Standard_Death");
334 gamemode.winScreen = Resources.FindObjectsOfTypeAll<UI_ScoreScreen>()

337 _gamemodeCache[cacheKey] = gamemode;
338 progress?.Report(1f);

351 public Sprite LoadPngAsSpriteRelative(string relativePngPath)
352 {
353 var fullPath = Path.Combine(_assemblyFolder, relativePngPath);
354 if (!File.Exists(fullPath))
355 {
356 WKLog.Error($"[AssetService] PNG not found at: {fullPath}");
357 return null;
358 }
359 var data = File.ReadAllBytes(fullPath);
360 var tex = new Texture2D(2, 2, TextureFormat.RGBA32, false);
361 if (!tex.LoadImage(data))
362 {
363 WKLog.Error($"[AssetService] Failed to load image: {relativePngPath}");
364 return null;
365 }
366 var sprite = Sprite.Create(tex, new Rect(0,0,tex.width,tex.height), new Vector2(0.5f,0.5f));
367 sprite.name = Path.GetFileNameWithoutExtension(relativePngPath);
368 return sprite;
369 }
